{{- $csvPath := .Get "csv" -}} {{- $csvContent := readFile (printf "static/%s" $csvPath) -}} {{- $lines := split (trim $csvContent "\n\r") "\n" -}} {{- $header := split (index $lines 0) "," -}} {{- $rows := slice -}} {{- range $i, $line := after 1 $lines -}} {{- $lineTrimmed := trim $line " \t\r\n" -}} {{- if gt (len $lineTrimmed) 0 -}} {{- $values := split $lineTrimmed "," -}} {{- $row := dict -}} {{- range $j, $col := $header -}} {{- $colName := trim (replace $col "\"" "") " " -}} {{- $val := "" -}} {{- if lt $j (len $values) -}} {{- $val = trim (replace (index $values $j) "\"" "") " " -}} {{- end -}} {{- $row = merge $row (dict $colName $val) -}} {{- end -}} {{- $rows = $rows | append $row -}} {{- end -}} {{- end -}}
